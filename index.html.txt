    <script>
        const noButton = document.getElementById('no');
        const yesButton = document.getElementById('yes');
        const successMessage = document.getElementById('success');
        const failMessage = document.getElementById('fail');

        // Fungsi untuk membuat tombol "No" bergerak acak
        function moveNoButton() {
            const x = Math.random() * (window.innerWidth - 100);
            const y = Math.random() * (window.innerHeight - 100);
            noButton.style.left = x + 'px';
            noButton.style.top = y + 'px';
        }

        // Event listener untuk tombol "No" (bergerak saat mouseover atau click)
        noButton.addEventListener('mouseover', moveNoButton);
        noButton.addEventListener('click', function() {
            moveNoButton();
            failMessage.style.display = 'block';
            setTimeout(() => failMessage.style.display = 'none', 2000); // Sembunyikan pesan setelah 2 detik
        });

        // Event listener untuk tombol "Yes"
        yesButton.addEventListener('click', function() {
            successMessage.style.display = 'block';
            noButton.style.display = 'none'; // Sembunyikan tombol "No" setelah "Yes" diklik
        });
    </script>
